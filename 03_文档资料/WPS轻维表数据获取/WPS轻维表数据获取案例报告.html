程序端调用方式（三选一） 
# PowerShell 调用 WPS AirScript Webhook
$webhookUrl = "https://365.kdocs.cn/api/v3/ide/file/{文件ID}/script/{脚本ID}/sync_task"
$token      = "你的AirScript令牌"

$response = Invoke-RestMethod `
  -Uri    $webhookUrl `
  -Method POST `
  -Headers @{"Content-Type"="application/json"; "AirScript-Token"=$token} `
  -Body   '{"Context":{"argv":{}}}' `
  -TimeoutSec 120

# 提取结果
$result = $response.data.result
Write-Host "获取成功: $($result.success), 总条数: $($result.total)"

# 保存 JSON（用 utf8 编码，Python 读取用 utf-8-sig）
$result.data | ConvertTo-Json -Depth 5 | Out-File "output.json" -Encoding utf8
import requests, json

webhook_url = "https://365.kdocs.cn/api/v3/ide/file/{文件ID}/script/{脚本ID}/sync_task"
token       = "你的AirScript令牌"

headers = {
    "Content-Type":    "application/json",
    "AirScript-Token": token
}

response = requests.post(
    webhook_url,
    headers=headers,
    json={"Context": {"argv": {}}},
    timeout=120
)

result = response.json()
data   = result["data"]["result"]["data"]
total  = result["data"]["result"]["total"]
print(f"获取成功，共 {total} 条记录")

with open("output.json", "w", encoding="utf-8") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)
// AirScript 1.0 全量分页读取模板
const TARGET_SHEET_ID = 2989; // 替换为目标 SheetId

try {
  let allRecords = [];
  let offset = null;

  do {
    let params = { SheetId: TARGET_SHEET_ID, PageSize: 100 };
    if (offset !== null) params.Offset = String(offset); // Offset 必须是字符串！

    let result  = Application.Record.GetRecords(params);
    let records = Array.from(result.records || []); // Array.from 必须加！
    allRecords  = allRecords.concat(records);
    offset = result.offset ? String(result.offset) : null;

  } while (offset);

  // 提取字段（按实际修改）
  let summary = allRecords.map(r => {
    let f = r.fields;
    return {
      id:       r.id,
      单据号:   f['单据号'],
      业务部门: f['业务部门'],
      业务员:   f['业务员'],
      重量:     f['重量'],
      提单金额万: f['提单金额（万）']
    };
  });

  return { success: true, total: allRecords.length, data: summary };

} catch (e) {
  return { success: false, error: e.message };
}

踩坑记录（避免重复踩坑）
  • ⚠️
    GetSheets() 返回空对象 {} 必须用 Array.from(Application.Sheet.GetSheets()) 包裹，否则遍历时字段全为空
  • ⚠️
    GetRecords() Offset 参数类型错误 Offset 必须是字符串类型，用 String(offset) 转换，不能直接传数字
  • ⚠️
    Python 读取 JSON 乱码 PowerShell Out-File 默认输出 UTF-8 BOM，Python 读取用 encoding='utf-8-sig'
  • ⚠️
    Webhook 请求超时 数据量大时脚本执行慢，设置 TimeoutSec 120 或更大
  • ⚠️
    AirScript 版本选错 必须选 1.0 版本，2.0 版本 API 不兼容，会报各种奇怪错误
  • ⚠️
    records 数组无法 map/forEach GetRecords() 返回的 records 也需要 Array.from() 处理，否则报错
